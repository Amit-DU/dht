% \caption fixes
\ifx\abovecaptionskip\@undefined\newskip\abovecaptionskip\fi

\long\def\@makecaption#1#2{%
  \vskip\abovecaptionskip
  \sbox\@tempboxa{{\small\textbf{#1}: #2}}%
  \ifdim \wd\@tempboxa >\hsize
    {\small\textbf{#1}: #2}\par
  \else
    \global \@minipagefalse
    \hb@xt@\hsize{\hfil\box\@tempboxa\hfil}%
  \fi
  \vskip\belowcaptionskip}


\def\@maketitle{%
  \newpage
  \begin{center}%
  \let \footnote \thanks
\expandafter\ifx\csname acmdescription\endcsname\relax
  \null
\else
  {\setbox0\hbox{\vbox{\begin{flushright}%
   \begin{tabular}[t]{r@{}}%
    \acmdescription
    \\ \noalign{\vskip0.25in}%
  \end{tabular}\end{flushright}\null}}\ht0=0pt\dp0=0pt\box0%
  }%
\fi
    \vskip -.25em%
    {\LARGE \@title \par}%
    \vskip 1.25em%
    {\large
      \lineskip .5em%
      \begin{tabular}[t]{c}%
        \@author
      \end{tabular}\par}%
  \end{center}%
  \par
  \vskip 1.5em}

\makeatother
